True values

beta =  c(3, 1.5, 0, 0, 2, 0, 0, 0)
sigma = 3
p = length(beta)
rho = 0.5
# covariance matrix
Sigma = matrix(NA, p, p)
for (i in 1:p){
  for (j in 1:p){
    Sigma[i,j] = sigma*rho^(abs(i-j))
  }
}
mu = rep(0,p)

Estimator for given sample size

Estimator<-function(n,a,b,seed_value){
  
  set.seed(seed_value)
  #generate design matrix X
  X = rmvnorm(n, mu, Sigma)
  
  #check for logistic regression 
  z = X%*%beta #matrix multiplication
  prob = exp(z)/(1+exp(z))
  #generate sample from binom
  y = rbinom(n,1,prob)
  
  ## MLE solution
  
  #simple logit with fisher's scoring MLE
  
  Data = cbind.data.frame(y,X)
  
  time_logit_start = Sys.time()
  fit_logit = glm(y~.-1, data = Data, family = binomial(link="logit"))
  time_logit_end = Sys.time()
  
  beta_hat_logit = coef(fit_logit)
  rmse_b_logit = sqrt(mean((beta_hat_logit - beta)^2))
  
  
  #Jacobi solution -----------------------------------------------------
  m = 1
  
  time_jacobi_start = Sys.time()
  beta_hat_jacobi = solve(t(X)%*%X)%*%t(X)%*%log((y+a)/(b+m-y))
  time_jacobi_end = Sys.time()
  
  rmse_b_jacobi = sqrt(mean((beta_hat_jacobi - beta)^2))
  
  return(list(rmse_b_logit=rmse_b_logit
              ,rmse_b_jacobi=rmse_b_jacobi))
  
}

Inconsistent estimator when \(a=b=1/2\)

Range of sample size

n_max=5000
n_start=200
res<-matrix(NA,ncol=7,nrow=length(seq(n_start,n_max,by=10)))
r<-1

Run the Monte Carlo Simulation

for(n in seq(n_start,n_max,by=200)){
  rmse_beta_logit<-rep(NA,No.of.Datasets)
  rmse_beta_Jacobi<-rep(NA,No.of.Datasets)
  for (k in 1:No.of.Datasets){
    
    fit = Estimator(n,a=1/2,b=1/2,seed_value=seed_str[k])
    
    rmse_beta_Jacobi[k]=fit$rmse_b_jacobi
    rmse_beta_logit[k]=fit$rmse_b_logit
  }
  
  res[r,]<-c(n
             ,quantile(rmse_beta_Jacobi
                       ,probs = c(0.025,0.5,0.975))
             ,quantile(rmse_beta_logit
                       ,probs = c(0.025,0.5,0.975))
             
  )
  
  r<-r+1
  if(n%%25==0) cat("n = ",n,"\n")
}
